> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add inline citation rendering with collapsible sources block, image lightbox, and twin profile management (avatar upload + name editing).

**Architecture:** Custom remark plugin replaces `[source:N]` markers with superscript React components during Markdown rendering. Frontend fetches profile on mount with env-var fallback.

import {
  FileText,
  FileType,
  Globe,

interface SourceIconInfo {
  icon: LucideIcon;
  iconName: string;
  color: string;
}

const SOURCE_ICON_MAP: Record<string, SourceIconInfo> = {
  pdf: { icon: FileText, iconName: "FileText", color: "#ef4444" },
  docx: { icon: FileText, iconName: "FileText", color: "#6b7280" },
  markdown: { icon: FileType, iconName: "FileType", color: "#6b7280" },
  txt: { icon: FileType, iconName: "FileType", color: "#6b7280" },
  html: { icon: Globe, iconName: "Globe", color: "#3b82f6" },
  image: { icon: ImageIcon, iconName: "ImageIcon", color: "#10b981" },
  audio: { icon: Headphones, iconName: "Headphones", color: "#f59e0b" },
  video: { icon: Video, iconName: "Video", color: "#a855f7" },
};

const DEFAULT_ICON: SourceIconInfo = {
  icon: FileText,
  iconName: "FileText",
  color: "#6b7280",
};

export function getSourceIcon(sourceType: string): SourceIconInfo {
  return SOURCE_ICON_MAP[sourceType] ?? DEFAULT_ICON;
}

- [ ] **Step 3: Implement remark-citations.ts**

The plugin walks the Markdown AST (mdast), finds text nodes containing `[source:N]`, splits them, and inserts inline HTML nodes that ReactMarkdown renders as React elements.

```typescript
// frontend/src/lib/remark-citations.ts
import type { Root, PhrasingContent, Text } from "mdast";
import type { Plugin } from "unified";
import type { CitationResponse } from "@/types/chat";

const CITATION_RE = /\[source:(\d+)\]/g;

interface RemarkCitationsOptions {
  citations: CitationResponse[];
}

export const remarkCitations: Plugin<[RemarkCitationsOptions], Root> = (options) => {
  const { citations } = options;
  const maxIndex = citations.length;

  return (tree) => {
    if (maxIndex === 0) return;
    visitText(tree);
  };

  function visitText(node: Root | PhrasingContent | { children?: unknown[] }): void {
    if (!("children" in node) || !Array.isArray(node.children)) return;

    const newChildren: unknown[] = [];
    let changed = false;

    for (const child of node.children) {
      if (child.type === "text") {
        const parts = splitTextNode(child as Text);
        if (parts.length > 1 || parts[0] !== child) {
          newChildren.push(...parts);
          changed = true;
          continue;
        }
      } else {
        visitText(child as PhrasingContent);
      }
      newChildren.push(child);
    }

    if (changed) {
      (node as { children: unknown[] }).children = newChildren;
    }
  }

  function splitTextNode(node: Text): PhrasingContent[] {
    const parts: PhrasingContent[] = [];
    let lastIndex = 0;
    let match: RegExpExecArray | null;

    CITATION_RE.lastIndex = 0;
    match = CITATION_RE.exec(node.value);
    while (match !== null) {
      const index = Number(match[1]);

      if (index < 1 || index > maxIndex) {
        match = CITATION_RE.exec(node.value);
        continue;
      }

      if (match.index > lastIndex) {
        parts.push({ type: "text", value: node.value.slice(lastIndex, match.index) });
      }

      parts.push({
        type: "html",
        value: `<sup><button type="button" class="citation-ref" data-citation-index="${index}" aria-label="${index}">${index}</button></sup>`,
      } as unknown as PhrasingContent);

      lastIndex = match.index + match[0].length;
      match = CITATION_RE.exec(node.value);
    }

    if (parts.length === 0) return [node];

    if (lastIndex < node.value.length) {
      parts.push({ type: "text", value: node.value.slice(lastIndex) });
    }

    return parts;
  }
};
```

**Integration note:** This plugin inserts `html` type mdast nodes. For ReactMarkdown to render them as actual HTML (not escaped text), `rehype-raw` MUST be added to the rehype plugin chain. This is done in Task 7 when integrating into MessageBubble. The remark plugin tests (above) use ReactMarkdown without sanitize, which works because react-markdown v10+ with `rehype-raw` processes HTML nodes correctly.

- [ ] **Step 4: Note on rehype-raw requirement**

The remark plugin produces `html` mdast nodes. ReactMarkdown requires `rehype-raw` to parse these into real HTML elements (without it, they render as escaped text). The `rehypeSanitize` config must also be extended to allow `sup` and `button[class, data-citation-index, aria-label, type]`. Both are configured in Task 7 when integrating into MessageBubble.

```typescript
// frontend/src/components/CitationsBlock/CitationsBlock.tsx
import { useState } from "react";
import { getSourceIcon } from "@/lib/source-icons";
import { strings } from "@/lib/strings";
import type { CitationResponse } from "@/types/chat";
import "./CitationsBlock.css";

interface CitationsBlockProps {
  citations: CitationResponse[];
  onImageClick?: (url: string, title: string) => void;
  highlightIndex?: number | null;
}

export function CitationsBlock({ citations, onImageClick, highlightIndex }: CitationsBlockProps) {
  const [expanded, setExpanded] = useState(false);

  if (citations.length === 0) return null;

  const count = citations.length;
  const label = strings.sourcesCount(count);

  return (
    <div className="citations-block">
      <button
        type="button"
        className="citations-block__toggle"
        onClick={() => setExpanded((prev) => !prev)}
        aria-expanded={expanded}
      >
        <span className="citations-block__arrow">{expanded ? "\u25BC" : "\u25B6"}</span>
        <span>{label}</span>
      </button>

      {expanded ? (
        <div className="citations-block__list" role="list">
          {citations.map((citation) => (
            <CitationItem
              key={citation.index}
              citation={citation}
              onImageClick={onImageClick}
              highlighted={highlightIndex === citation.index}
            />
          ))}
        </div>
      ) : null}
    </div>
  );
}

function CitationItem({
  citation,
  onImageClick,
  highlighted,
}: {
  citation: CitationResponse;
  onImageClick?: (url: string, title: string) => void;
  highlighted?: boolean;
}) {
  const { icon: Icon, color } = getSourceIcon(citation.source_type);
  const isImage = citation.source_type === "image" && citation.url != null;
  const isOnline = citation.url != null && !isImage;

  const handleClick = () => {
    if (isImage && citation.url) {
      onImageClick?.(citation.url, citation.source_title);
    }
  };

  return (
    <div
      className={`citations-block__item ${highlighted ? "citations-block__item--highlighted" : ""}`}
      role="listitem"
      data-citation-index={citation.index}
    >
      <span className="citations-block__num">{citation.index}</span>
      <Icon className="citations-block__icon" size={16} style={{ color }} />
      <div className="citations-block__info">
        {isOnline ? (
          <a
            className="citations-block__title citations-block__title--link"
            href={citation.url!}
            target="_blank"
            rel="noopener noreferrer"
          >
            {citation.source_title}
          </a>
        ) : isImage ? (

# backend/app/api/profile.py
from __future__ import annotations

import uuid

from fastapi import APIRouter, Depends, HTTPException, UploadFile
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.api.profile_schemas import AvatarUploadResponse, ProfileUpdateRequest, TwinProfileResponse
from app.core.constants import DEFAULT_AGENT_ID
from app.db.models.core import Agent
from app.db.session import get_session
from app.services.storage import StorageService

ALLOWED_IMAGE_TYPES = {"image/jpeg", "image/png", "image/webp", "image/gif"}
MAX_AVATAR_SIZE = 2 * 1024 * 1024  # 2MB

chat_router = APIRouter(prefix="/chat", tags=["chat"])
admin_router = APIRouter(prefix="/admin/agent", tags=["admin"])


async def _get_agent(db: AsyncSession) -> Agent:
    result = await db.execute(select(Agent).where(Agent.id == DEFAULT_AGENT_ID))
    agent = result.scalar_one_or_none()
    if agent is None:
        raise HTTPException(status_code=404, detail="Agent not found")
    return agent


@chat_router.get("/twin", response_model=TwinProfileResponse)
async def get_twin_profile(db: AsyncSession = Depends(get_session)) -> TwinProfileResponse:
    agent = await _get_agent(db)
    return TwinProfileResponse(
        name=agent.name,
        has_avatar=agent.avatar_url is not None,
    )


@chat_router.get("/twin/avatar")
async def get_twin_avatar(
    request: Request,
    db: AsyncSession = Depends(get_session),
) -> Response:
    agent = await _get_agent(db)
    if not agent.avatar_url:
        raise HTTPException(status_code=404, detail="No avatar set")
    storage: StorageService = request.app.state.storage_service
    content = await storage.download(agent.avatar_url)
    ext = agent.avatar_url.rsplit(".", 1)[-1].lower()
    media_type = {"png": "image/png", "jpg": "image/jpeg", "jpeg": "image/jpeg",
                  "webp": "image/webp", "gif": "image/gif"}.get(ext, "image/png")
    return Response(content=content, media_type=media_type)


@admin_router.put("/profile", response_model=TwinProfileResponse)
async def update_profile(
    payload: ProfileUpdateRequest,
    db: AsyncSession = Depends(get_session),
) -> TwinProfileResponse:
    agent = await _get_agent(db)
    agent.name = payload.name
    await db.commit()
    await db.refresh(agent)
    return TwinProfileResponse(
        name=agent.name,
        has_avatar=agent.avatar_url is not None,
    )


@admin_router.post("/avatar", response_model=AvatarUploadResponse)
async def upload_avatar(
    file: UploadFile,
    request: Request,
    db: AsyncSession = Depends(get_session),
) -> AvatarUploadResponse:
    if file.content_type not in ALLOWED_IMAGE_TYPES:
        raise HTTPException(status_code=422, detail=f"Invalid image type: {file.content_type}")

    content = await file.read()
    if len(content) > MAX_AVATAR_SIZE:
        raise HTTPException(status_code=422, detail="File too large (max 2MB)")

    agent = await _get_agent(db)
    storage: StorageService = request.app.state.storage_service

    ext = (file.filename or "avatar.png").rsplit(".", 1)[-1] if file.filename else "png"
    object_key = f"agents/{agent.id}/avatar/{uuid.uuid4().hex}.{ext}"
    await storage.upload(object_key, content, content_type=file.content_type or "image/png")

    # Delete old avatar if exists (best-effort cleanup)
    if agent.avatar_url:
        try:
            await storage.delete(agent.avatar_url)
        except Exception:
            pass

    # Store object key (not URL) — browser accesses via /api/chat/twin/avatar proxy
    agent.avatar_url = object_key
    await db.commit()
    await db.refresh(agent)
    return AvatarUploadResponse(has_avatar=True)


@admin_router.delete("/avatar", response_model=AvatarUploadResponse)
async def delete_avatar(
    request: Request,
    db: AsyncSession = Depends(get_session),
) -> AvatarUploadResponse:
    agent = await _get_agent(db)
    if agent.avatar_url:
        storage: StorageService = request.app.state.storage_service
        try:
            await storage.delete(agent.avatar_url)
        except Exception:
            pass
        agent.avatar_url = None
        await db.commit()
        await db.refresh(agent)
    return AvatarUploadResponse(has_avatar=False)
```
